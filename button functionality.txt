final Button b1;
        final ToggleButton pencil;
        final RadioGroup radgroup;

        b1 = findViewById(R.id.butt1);
        radgroup = findViewById(R.id.radiogroup1);
        pencil = findViewById(R.id.toggleButton2);

        b1.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {

                if(radgroup.getCheckedRadioButtonId() == R.id.radioBut1) {  // if radio button 1 is checked when button1 clicked then change text to '1'
                    Resources res = getResources();
                    String testing = res.getString(R.string.button1);
                    b1.setText(testing);
                } else if(radgroup.getCheckedRadioButtonId() == R.id.radioBut2) {
                    Resources res = getResources();
                    String testing = res.getString(R.string.buttonon);
                    b1.setText(testing);
                }

                if(pencil.isChecked()) {
                    Resources res = getResources();
                    String testing = res.getString(R.string.buttonon);
                    b1.setText(testing);
                    b1.setTextColor(Color.GREEN);
                } else {
                    b1.setTextColor(Color.BLACK);
                }
            }
        });
